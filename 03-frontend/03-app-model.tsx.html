# فایل app-model.tsx

نکات مربوط به کد

فریم ورک برای اینکه مدل از دامین شما داشته باشد ، به یک کلاس نیاز دارد ، این کلاس شامل

  • چک کردن مجوز ها – متد checkPermission
  • نمایش منو ها صرفا جهت استفاده در صفحه MasterPage -متد getMenuItems()
  • تعیین کامپونت MasterPage - متد defaultMasterPage()

import { DefaultMasterPage } from "./master-page";

import { Menu, IAppModel } from '@organic-ui';

export default class AppModel implements IAppModel {
    static permissions: any[];
    static permissionsTable: { [key: number]: true };

    checkPermission(permissionKey) {
        if (!permissionKey) return true;
        if (permissionKey instanceof Array)
            return permissionKey.reduce((result, key) => result || this.checkPermission(key), false);
        if (!AppModel.permissions) return true;
        AppModel.permissionsTable = AppModel.permissionsTable || Object.assign({}, ...(AppModel.permissions.map(p => p.permission).map(key => ({ [key]: true }))))
        if (AppModel.permissionsTable && AppModel.permissionsTable['IsAdministrator']) return true;
        return AppModel.permissionsTable[permissionKey];
    }

    getMenuItems() {
        return [
            {
                menu: new Menu(100, 'customers-management', "", null, 'fa-money', null, true, 0)
            },
            {
                menu: new Menu(101, 'customers', '/view/customers', null, 'fa-user', null, false, 100)
            },
            {
                menu: new Menu(102, 'related-to-customers', '/view/people/related-to-customers', null, 'fa-address-card', null, false, 100)
            },
            {
                menu: new Menu(200, 'resources-management', '', null, 'fa-simplybuilt', null, true, 0)
            },
            /* {
                 menu: new Menu(12, 'tickets', '/view/tickets', null, 'fa-ticket', null, false, 0)
             },*/{
                menu: new Menu(203, 'products', '/view/products', null, 'mdi-flask', null, false, 200)
            },
            {
                menu: new Menu(201, 'install-experts', '/view/people/experts', null, 'mdi-account-check', null, false, 200)
            }, {
                menu: new Menu(202, 'user-accounts', '/view/user-accounts', null, 'fa-users', null, false, 200)
            },
            {
                menu: new Menu(300, 'reports', '', null, 'fa-print', null, true, 0)
            },
            {
                menu: new Menu(301, 'licenses-report', '/view/reports/licenses', null, 'mdi-cloud-print-outline', null, false, 300)
            }
        ];

    }
    defaultMasterPage() {
        return DefaultMasterPage;
    }
}
