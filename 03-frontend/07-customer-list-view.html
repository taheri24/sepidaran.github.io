# پیاده سازی بوسیله reinvert - فقط حالت لیست

export const customers = reinvent<CustomerDTO>('frontend:crud', { actions:customersApi, options })
    .listView(({ binding }) => <>
        <DataList>
            <Field accessor={binding.id} >
                <TextField />
            </Field>
            <Field accessor={binding.customerName}  >
                <TextField />
            </Field>
        </DataList>
    </>
    ).done({ moduleId: 'employees' });
