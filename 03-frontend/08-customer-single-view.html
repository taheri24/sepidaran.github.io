# employeeLogics - عملیات اعطای کارت ها

قوائد مربوط اعطای کدفعال سازی

  • برای اعطای کدفعال سازی حتما باید درخواست ذخیره شود و بعد از آن تولید کد فعال امکان پذیر است
  • اگر کد فعال سازی تولید شده باشد - امکان حذف وجود نخواهد داشت
  • برای ویرایش ، اگر کدفعال سازی تولید شده باشد محدودیت وجود دارد
  • اعطای کد فعال و ثبت آن توسط سمت سرور با یک درخواست POST انجام میشود
const employeeLogics = {
    licenseActionExecute(funcKey, row) {
        if (row.activationKey) {
            if (funcKey == 'remove')
                return Utils.failCallout('generatedCode-is-readonly');
            if (funcKey == 'generateActivationCode' && !row.id)
                return Promise.resolve(Utils.failCallout('activation-key-after-save'));
            if (funcKey == 'edit') {
                return {
                    isFieldReadonly: fld => ['gener'].includes(Field.getAccessorName(fld))
                } as OrganicUi.DataForm.FormSettings;
            }
        }
    }
}

# پیاده سازی بوسیله ES6-Class

نکات مربوط به کد

  • @SelfBind : این تابع یک Decorator است مربوط برنامه نویسی AOP - و برای حل کردن مشکل prototype در هنگام تعیین رویداد می باشد - نیازی به bind تابع در هنگام اختصاص آن نیست
  • @templatedView : بوسیله این decorator شما می توانید برای کامپونت مورد نظرتان الگو تعیین کنید
  • کامپونت Board
    • موجودیت کارت ها در بانک اطلاعاتی ذخیره نمیشود - برای آن sync درنظر گرفت
    • تبدیل اطلاعات DTO به کارت ها در تابع getCards انجام میشود
    • اعمال کارت ها به DTO در متد onBoardDataChange
  • بوسیله رویداد onCaptureDataForm اطلاعات فرم به فیلد customer انتقال پیدا میکند
 



















































































































































import { BaseComponent, routeTable, DataPanel, Field, Utils, ComboBox, TextField, Board, SelfBind, DataListPanel, HiddenField } from "@organic-ui";
import { templatedView, openBindingSource, BindingHub, } from "@reinvent";
import { options } from "./customers-view";
import { customersApi } from "./api";
import { CustomerDTO } from "./entities";
import licenseCard from "./cards/license-card";
import personCard from "./cards/person-card";

enum CustomerType {
    Organization = 1,
    RealPerson = 2
}


const employeeLogics = {
    licenseActionExecute(funcKey, row) {
        if (row.activationKey) {
            if (funcKey == 'remove')
                return Utils.failCallout('generatedCode-is-readonly');
            if (funcKey == 'generateActivationCode' && !row.id)
                return Promise.resolve(Utils.failCallout('activation-key-after-save'));
            if (funcKey == 'edit') {
                return {
                    isFieldReadonly: fld => ['gener'].includes(Field.getAccessorName(fld))
                } as OrganicUi.DataForm.FormSettings;
            }
        }
    }
}
interface IState {

}
class CustomerSingleView extends BaseComponent<any, IState>{
    binding: BindingHub<CustomerDTO>;
    customer: CustomerDTO;
    cards: Array<any>;
    boardKey: number = 0;
    static mapItemsWithCardType(array: Array<any>, cardType: string) {
        if (array instanceof Array)
            return array.map(item => ({ ...item, cardType }));
        return [];
    }

    cardToEntity(card) {
        const { id, _cardIndex, ...cardOtherProps } = card;
        return { ...cardOtherProps, id: card._new ? undefined : id, index: _cardIndex, ...(card.data || {}) }
    }
    @SelfBind()
    onBoardDataChange({ cards, lanes }) {
        if (!this.customer) return;

        this.customer.lanes = lanes.map((lane, index) => ({ ...lane, index }));
        const _cards = cards.map((c, _cardIndex) => ({ ...c, _cardIndex }));
        this.customer.licenses = _cards.filter(({ cardType }) => cardType == 'license').map(this.cardToEntity);
        //this.customer.people = _cards.filter(({ cardType }) => cardType == 'people').map(this.cardToEntity);
        
        this.boardKey = +new Date();
    }

    getCards(customer: CustomerDTO) {
        if (!customer) return [];

        const cards = [].concat(
            //CustomerSingleView.mapItemsWithCardType(customer.people, 'people'),
            CustomerSingleView.mapItemsWithCardType(customer.licenses, 'license')
        ).sort((a, b) => (a.index || 0) - (b.index || 0));
        return cards;
    }
    @templatedView('singleView', { options, actions: customersApi })
    renderContent() {
        this.binding = this.binding || openBindingSource<CustomerDTO>();

        const { binding } = this;
        const people = binding.people[0];
        if (!this.customer) {
            this.repatch({}, null, 100);
        }

            Object.assign(window,{dto:this.customer});
        return <>
            <DataPanel header="primary-fields" primary className="half-column-fields" >

                <Field onCaptureDataForm={customer => this.customer = customer} accessor={binding.customerName} >
                    <TextField />
                </Field>
                <Field accessor={binding.customerType} defaultValue={1} >
                    <ComboBox items={Utils.enumToIdNames(CustomerType)} />
                </Field>


                <Field accessor={binding.phone} >
                    <TextField />
                </Field>
                <Field accessor={binding.address}   >
                    <TextField />
                </Field>
                <Field accessor={binding.comments} className="full-width" >
                    <TextField />
                </Field>

            </DataPanel>
            <DataPanel header="licenses" className="half-column-fields" >

                {this.customer &&
                    <Board
                        accquireLaneId={Utils.uuid}
                        cardMapping={{ cardType: 'cardType', id: 'id', laneId: 'laneId' }}
                        lanes={() => this.customer.lanes}
                        key={this.boardKey}
                        cards={this.getCards.bind(this, this.customer)}
                        onDataChange={this.onBoardDataChange.bind(this)}
                        cardTypes={{ 'license': licenseCard, 'people': personCard }} />

                }
            </DataPanel>
            <DataListPanel header="people" accessor={binding.people} formMode="modal"
                contentClassName="small-fields" singularName="person"
                onActionExecute={employeeLogics.licenseActionExecute}
                dataListHeight={300} dataFormHeight={500}  >
                <Field accessor={people.fullName} style={{ minWidth: '49%' }} columnProps={{ minWidth: 300, maxWidth: 300 }}  >
                    <TextField />
                </Field>
                <Field accessor={people.role} style={{ minWidth: '49%' }} columnProps={{ minWidth: 100, maxWidth: 100 }}  >
                    <TextField />
                </Field>
                <Field accessor={people.peopleType} dataEntryOnly defaultValue={2}>
                    <HiddenField />
                </Field>
                <Field accessor={people.mobile}  >
                    <TextField placeholder="09351234567" dir="ltr" />
                </Field>
                <Field accessor={people.phone} dataEntryOnly>
                    <TextField placeholder="021xxxxx" dir="ltr" />
                </Field>
                <Field accessor={people.email}  >
                    <TextField placeholder="email@host.com" dir="ltr" />
                </Field>
                <Field accessor={people.telegramId}   >
                    <TextField placeholder="@id" dir="ltr" />
                </Field>
            </DataListPanel>
        </>
    }

}

routeTable(options.routeForSingleView, CustomerSingleView);
