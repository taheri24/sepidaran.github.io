# تنظیم BaseURL

در توسعه نرم افزار ها معمولا پیش میاد که بخواهیم به دامین دیگری وصل بشیم . به این مکانزیسم Cross-Origin Resource Sharing (CORS) میگویند که معمولا در هنگام توسعه Development لازم هستش معمولا ما به دات نت یا تکنولوژی دیگری وصل می شویم . طبعا پورت آنها فرق می کند، البته ما این کار امکان را می دهیم ، که پروسه Build در فولدر دلخواه بیاد شود ، مشکل CORS شما در این صورت نخواهید داشت ولی باید سرور شما قابلیت هندل فایل ها به خوبی داشته باشد . مثل همه محیط های توسعه JavaScript ، سازگارترین گزینه برای فرانت اند، Nodejs میباشد که در پروژه سپیداران از سرور Nodejs استفاده شده است. در کد زیر از localStoage که همان حافظه دراز مدت مرورگر استفاده شده است . معمولا در حالت Production نیازی به تنظیم BaseURL نیست ، ضمنا می توانید به وسیله آی پی به کامپیوتر سایر برنامه نویسان شرکت متصل شوید ، و برای مثال http://192.168.1.30:51368 یعنی آدرس کامپیوتر همکار شما در شبکه

const baseURL = (localStorage.getItem('baseURL')) || '' //|| `http://${location.hostname}:51368/`;

# تنظیمات - webApiClientSettings

  • برای اتصال ایمن X-Token تنظیم میکنیم
  • برای ساپورت CORS ، HTTP-Headers را باید به شکل زیر تنظیم کنیم
  • baseURL باید دارای getter و setter باشد
  • برای اعتبار سنجی ، از تابع validateStatus استفاده میکنیم
  • در نهایت برای آنها title نیز تنظیم میکنیم . که کاربرد Debug دارد
  • این فریم ورک برای ارتباط بین سرور و کلاینت از Axois بهره مند شده است
  • شما می توانید تنظیمات بسیار زیادی داشته باشید ، برای توضیحات بیشتر به مستندات Axois مراجعه کنید.
const webApiClientSettings: OptionsForRESTClient = () => {
    const token = localStorage.getItem('token');
    return {
        title: 'Web API',
        baseURL,//: baseURL || `http://192.168.1.66:51368/`,//  `http://${location.hostname}:51368/`,
        setBaseURL: baseURL => localStorage.setItem('baseURL', baseURL),
        headers: {
            'X-Token': token,
            'Access-Control-Allow-Origin': '*'
        },
        validateStatus: status => {
            if (status == 401) switchToLoginPage();
            return (status >= 200) && (status <= 299);
        }
    };
}

# پیاده سازی API برای CRUD

برای توسعه سریعتر ساختار CRUD در API ما در یک تابع Factory پیاده کرده ایم که صرفا با یـــک خط خود گروهی از توابعی در ایجاد شود

function crudApiFactory<T>(rootApiPath: string | T[]): IActionsForCRUD<T> ;
function setupPredefineApi<T>(name, apiPathOrData, overrideMethods?: Partial<IActionsForCRUD<T>>): IActionsForCRUD<T> {
    return reinvent.predefinesForApi[name] =
        Object.assign(crudApiFactory<T>(apiPathOrData) as any, overrideMethods || {});
}

# ساختار IActionsForCRUD

ساختار CRUD در قالب یک interface مشخص شده است که خروجی توابع Promise می باشد

export interface IActionsForCRUD<TDto> {
    // CRUD
    create: (dto: TDto) => Promise<any>;
    read: (id: any) => Promise<TDto>;
    readList: (params: IAdvancedQueryFilters) => PromisedResultSet<TDto>;
    readByIds: (ids: any[]) => PromisedResultSet<TDto>;
    update: (id: any, dto: TDto) => Promise<any>;
    deleteList: (hid: any[]) => Promise<any>;
    //  optional methods
    mapFormData?: (dto: TDto) => TDto;
    beforeSave?: (dto: TDto) => TDto;
    readListByMode?: Map<string, (params: IAdvancedQueryFilters) => PromisedResultSet<TDto>>;
    getUrlForSingleView?(id: string): string;
    validate?: (data: any) => IDataFormAccessorMsg[];
    getText?: (dto: TDto) => string;
    getId?: (dto: TDto) => any;
    getPageTitle?: (dto: TDto) => string;
    onFieldWrite?: (key: string, value, dto: TDto) => void
}
