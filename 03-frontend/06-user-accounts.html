# ماژول کاربران

  • پیچیدگی این ماژول در سطح متوسط است، ماژول محصولات بسیار ساده بود
  • در حال این کد دارای پیچیدگی متوسطی هست که باید دقت بیشتری کرد

نکات مربوط به کد

  • تعیین مجوز در قسمت دوم options
  • افزودن فیلد کمکیpasswordHash2 با تعریف یک interface جدید
  • اعتبار سنجی با کد دلخواه در رویداد onErrorCode
  • دارای جزئیات برای مجوز های مربوط به کاربر
  • استفاده از ساختار داده Permission(enum) در ComboBox در قسمت پیاده سازی

# پیاده سازی بوسیله reinvert

این کد شامل single-view, list-view می باشد و هردو ماژول در یک جا تعریف شده است

const userAccountApi = reinvent.predefinesForApi['@user-accounts'];
interface _UserAccountDTO extends UserAccountDTO {
	passwordHash2: string;
}
reinvent<_UserAccountDTO>('frontend:crud', { options, actions:userAccountApi })
	.singleView
	(({ binding, getData }) => <>
		{/*Part(single-view,DataPanel,0147b98df369f61783d61fecc233fae6)*/}
		<DataPanel header="primaryFields" className="half-column-fields">
			<Field accessor={binding.userName} className="full-width">
				<TextField />
			</Field>
			<Field accessor={binding.passwordHash} >
				<TextField type="password" autoComplete="new-password" autoCorrect="false" />
			</Field>

			<Field onErrorCode={retypedPassword => {
				const data = getData();
				return data && data.passwordHash == retypedPassword ? '' : 'retyped-password-invalid'
			}} accessor={binding.passwordHash2} >
				<TextField type="password" autoComplete="new-password" />
			</Field>
		</DataPanel>
		<DataListPanel accessor={binding.permissions} header="permissions" dataListHeight={300} contentClassName="full-width-fields"
			singularName="permission">
			<Field accessor={binding.permissions[0].permission} columnProps={{ minWidth: 400 }}>
				<ComboBox items={Utils.enumToIdNames(Permission)} />
			</Field>
		</DataListPanel>

	</>)
	.listView
	(({ binding }) => <>
		{/*Part(data-list,DataList,e36aa682f44d4685bf23a77f6625f73a)*/}
		<DataList>
			<Field accessor={binding.id}>
				<TextField />
			</Field>
			<Field accessor={binding.userName}>
				<TextField />
			</Field>

		</DataList>
	</>).done({ moduleId: 'user-accounts' })
