# فایل domain.tsx - ایندکس frontend

در webpack معمولا یک فایل ورودی وجود دارد ، این فایل به عنوان کد ورودی webpack می باشد – در واقع همه ماژول های دامین از اینجا import می شود . این فایل به سه قسمت تقسیم شده است .

  • قسمت اول : رجوع به تعاریف خودمان – ما چندین فایل مهم تعریف داریم که ساختار پروژه به آن اشاره شده است . فایل تعریف پسوند شان .d.ts می باشد – خط اول و خط دوم
  • وصل کردن ماژول های سیستم به فایل ورودی – ماژول های ورودی به صورت نسبی در کنار فایل دامین قرار دارند بسته به نوع ساختار مسیر import مشخص می شود
  • قسمت سوم : بعد از وصل کردن ماژول ها لازم است یکی از ماژولها نسبت صفحه جاری «لود» و render کنید، فریم ورک بوسیله تابع startApp ، این کار برایتان انجام می دهند لحظه ای باید آنرا فراخوانی کرد که کار لود صفحه تمام شده باشد
// PART I
/// <reference path="../dts/globals.d.ts" />
/// <reference path="../dts/organic-ui.d.ts" />
// PART II
import './frontend/api';
import './frontend/customers-single-view';
import './frontend/customers-view';
import './frontend/products-view';
import './frontend/people-view';
import './frontend/user-account-view';
import './frontend/reports/licenses-report';
import AppModel from './frontend/app-model';
import { startApp } from '@organic-ui';
import './frontend/ui-customization'; 
// PART III
window.addEventListener('DOMContentLoaded', () => startApp(new AppModel()), { once: true });
