# فایل Api.ts - ایندکس سرور

این فایل امکانات که در مسیر /api است مشخص میکند در واقع این واقع سایر ماژول های سیستم لود میکند این فایل به عنوان ورودی webpack در پروسه ایجاد فایل bundle سمت سرور می باشد

/// <reference path="./sepid-api-server.d.ts" />
import { setupApiRouter, mountApiRouter, webResult, setupReportService, setupMiddleware } from 'sepid-api-server'
import dataAccessLayer from "./data-access-layer";
import schemaLayer from './schema-layer';
import authModule, { checkTokenMiddle } from './auth';
import licenseModule from './license';
setupMiddleware(checkTokenMiddle);
mountApiRouter('/auth', authModule);
mountApiRouter('/license', licenseModule);
setupApiRouter('/customers', dataAccessLayer.customer, schemaLayer.customer);
setupApiRouter('/products', dataAccessLayer.product, schemaLayer.product);
setupApiRouter('/tickets', dataAccessLayer.ticket, schemaLayer.ticket);
const { people: peopleDA } = dataAccessLayer;
const peopleRouter = setupApiRouter('/people', peopleDA, schemaLayer.people);
peopleRouter.get('/related-to-customers', async req => webResult('query', peopleDA.dynamicQuery('ReadRelatedToCustomers', req, req.query)));
peopleRouter.get('/experts', async req => webResult('query', peopleDA.dynamicQuery('ReadExperts', req, req.query)));
setupApiRouter('/user-accounts', dataAccessLayer.userAccount, schemaLayer.userAccount);
// Report Service
const { report: reportDA } = dataAccessLayer;
setupReportService(reportDA, 'licenses');

# import - لود ماژول ها

اولین خط برنامه توابع و کلاس هایی از هسته سیستم لود می کند
sepid-api-server نام ماژول هسته سیستم است که توابع در فولدر dts تعریف شده است

# تنظیم Middleware

تنظیم پروسه «چک کردن توکن» در هر Request هایی که برای API فرخوانی می شود نیازمند یک middleware است البته middleware کاربرد های مختلفی دارد ازجمله Log,Monitor,موضوعات امنیتی,...

# تنظیم Router های مربوط به /api

تنظیم روتر ها بوسیله دو تابع mountApiRouterو setupApiRouter تنظیم می شود .

mountApiRouter('/auth', authModule);
mountApiRouter('/license', licenseModule);
setupApiRouter('/customers', dataAccessLayer.customer, schemaLayer.customer);
setupApiRouter('/products', dataAccessLayer.product, schemaLayer.product);
setupApiRouter('/tickets', dataAccessLayer.ticket, schemaLayer.ticket);

# تعریف ماژول گزارش گیری

عملیات گزارشگیری در هسته سیستم هندل می شود . فقط در سطح دامین سطح دسترسی به داده مشخص می شود کد آن نیز ساده می باشد

// Report Service
const { report: reportDA } = dataAccessLayer;
setupReportService(reportDA, 'licenses');
