# مقدمه سمت سرور

همان گونه که در بخش مقدمه آمده بود ، ما برای سمت سرور این برنامه از Nodejs بهره بردیم ، دنیای Nodejs تجربه بهتری از سمت سرور به برنامه نویسان ارائه داده است و در هنگام نگارش این سند ،در نسخه یازده می باشد بعد این همه سال این پروژه به بلوغ فوق العاده ای رسیده ، و هر لحاظ (امکانات-سرعت توسعه -کارایی -پایداری) از بالاترین استاندارد پیروی میکند تیم توسعه به دنبال یک فریم ورک مناسب برای سمت سرور بود و در نهایت در تیم توسعه تصمیم براین شد تجربه فریم ورک نیز در سمت سرور با این محیط برنامه نویسی داشته باشیم برای توسعه هسته فریم ورک از جاوا-اسکریپت ES8 و برای ایجاد دامین از TypeScript بهره بردیم فایل های TypeScript بوسیله Webpack به جاوا-اسکریپت(این بار برای سمت سرور) تبدیل می شود

  • Server/Core Written in Pure Javascript for Nodejs
  • Server/Domain Written in TypeScript

# قرار دادن برنامه در سرور - Deploy

برنامه های Nodejs برای اجرای بصورت ویندوز سرویس نیاز یک اجرا کننده سرویسی دارند ، یکی از متداول ترین آنها nssm می باشد ، اجرا این برنامه نیز در حالت ویندوز سرویس توسط همین برنامه انجام میشود

انجام تنظیمات در nssm بصورت خط فرمان می باشد برای سهولت کار می توانید از فایل زیر استفاده کنید

REM install-nodejs-service.cmd 
setlocal
cd /d %~dp0
 
SET CURRENTDIR=%cd%
SET nssm=%CURRENTDIR%\nssm.exe 
SET SERVICE_NAME=sepidaran_service
%nssm% stop %SERVICE_NAME% confirm
TIMEOUT   3

%nssm% remove %SERVICE_NAME% confirm

TIMEOUT   3
%nssm% install  %SERVICE_NAME% "%CURRENTDIR%\node.exe" "%CURRENTDIR%\src\index.js" --databaseConnection="%CURRENTDIR%\database-connection.json"   --serverPort=8081 
%nssm% set %SERVICE_NAME% AppDirectory "%CURRENTDIR%"
%nssm%  start  %SERVICE_NAME%
